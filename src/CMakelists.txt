#project name
PROJECT(fx_tools)

CMAKE_MINIMUM_REQUIRED(VERSION 3.8)

IF (NOT FX_BOOST_PATH)
	SET(FX_BOOST_PATH ../../../boost)
ENDIF ()

IF (NOT FX_BOOST_INCLUDE_PATH)
	SET (FX_BOOST_INCLUDE_PATH ${FX_BOOST_PATH})
	SET (Boost_INCLUDE_DIRS ${FX_BOOST_PATH})
ENDIF ()
IF (NOT EXISTS ${FX_BOOST_INCLUDE_PATH})
	MESSAGE(WARNING "boost not found!")
ENDIF ()

IF (NOT FX_BOOST_LIB_PATH)
	SET (FX_BOOST_LIB_PATH ${FX_BOOST_PATH}/stage/lib)
	SET (Boost_LIBRARIES ${FX_BOOST_PATH}/stage/lib)
ENDIF ()
IF (NOT EXISTS ${FX_BOOST_LIB_PATH})
	MESSAGE(WARNING "boost lib not found!")
ENDIF ()

MESSAGE("------------------------------boost------------------------------")
MESSAGE("boost include path : " ${FX_BOOST_INCLUDE_PATH})
MESSAGE("boost lib path : " ${FX_BOOST_LIB_PATH})

#SET (BOOST_ROOT ${FX_BOOST_PATH})

SET(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} /MTd")
SET(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} /MT")

SET(CMAKE_RUNTIME_OUTPUT_DIRECTORY ../fx_tools/bin)
SET(CMAKE_RUNTIME_OUTPUT_DIRECTORY_DEBUG ../fx_tools/bin)
SET(CMAKE_RUNTIME_OUTPUT_DIRECTORY_RELEASE ../fx_tools/bin)

ADD_SUBDIRECTORY(fx_duplicate_files)
